TABLES

CREATE TABLE CUSTOMER(
  CUST_ID VARCHAR(15) PRIMARY KEY,
  CUSTOMER_NAME	 VARCHAR(15) NOT NULL,
  PAN	 VARCHAR(10),
  DOB	 DATE,
  MOBILE INTEGER,
  EMAIL  VARCHAR(20),
  ALT_MOBILE INTEGER,
  ALT_EMAIL 	VARCHAR(20)
);

CREATE TABLE ACCOUNT(
  CUST_ID VARCHAR(15) ,
  ACCOUNT_ID VARCHAR(15) ,
  OPENING_DATE DATE NOT NULL,
  ACCOUNT_TYPE VARCHAR(15)  NOT NULL,
  BALANCE DOUBLE NOT NULL,
  PRIMARY KEY(CUST_ID, ACCOUNT_ID)
);

CREATE TABLE CUST_LOGIN(
CUST_ID VARCHAR(15),
USERNAME VARCHAR(15) PRIMARY KEY,
PASSWORD VARCHAR(15) NOT NULL,
LASTLOGIN TIMESTAMP,
LOCKED_STATUS BOOLEAN NOT NULL
);

CREATE TABLE TRANSACTIONS(
  TRANS_ID INTEGER PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 5000, INCREMENT BY 1),
  ACCOUNT_ID VARCHAR(15),
  AMOUNT DOUBLE NOT NULL,
  TRANS_TYPE VARCHAR(6),
  TRANS_TIME TIMESTAMP NOT NULL,
  REMARKS VARCHAR(20)
);

CREATE TABLE EMP_LOGIN(
EMP_ID VARCHAR(15) NOT NULL,
USERNAME VARCHAR(15) PRIMARY KEY,
EMPLOYEE_NAME	 VARCHAR(15) NOT NULL,
PASSWORD VARCHAR(15) NOT NULL,
LASTLOGIN TIMESTAMP NOT NULL
);
